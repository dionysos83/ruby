Zuweisung :
	variablen_name = "Dies ist ein String"
	#eine Zuweisung erfolgt über den Zuweisungsoperator '='
	#der Variablentyp (String, integer, time) erfolgt über die Eingabe

Zeichenkette (String):
	str = "Dies und 15 andere."
		#ein String wird erzeugt, in dem eine beliebige Anzahl von Zeichen (Zahlen, Buchstaben) innerhalb von Aführungszeichen (
		sowohl einfach als auch dopppelte) steht.
		Zusammengesetzte Zeichenkette:
			tag = "Freitag"
			str = "Heute ist " + tag
				#das +-Zeichen verbindet beide Strings

Ganze Zahl (Integer, Fixnum)
	zahl = 15

Float (Kommazahl)
	komma_zahl = 13.44
